335 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
336  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
337  SLICE99_ASSERT(comparator);
338 
339  if (lhs.len != rhs.len) {
340  return false;
341  }

343  for (size_t i = 0; i < lhs.len; i++) {
344  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
345  return false;
346  }
347  }

349  return true;
