313 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
314  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
315  SLICE99_ASSERT(comparator);
316 
317  if (lhs.len != rhs.len) {
318  return false;
319  }
320 
321  for (size_t i = 0; i < lhs.len; i++) {
322  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
323  return false;
324  }
325  }
326 
327  return true;
