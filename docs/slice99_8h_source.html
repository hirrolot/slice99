403 inline static SLICE99_WARN_UNUSED_RESULT bool
404 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
405  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
406  SLICE99_ASSERT(comparator);
407 
408  if (lhs.len != rhs.len) {
409  return false;
410  }
411 
412  for (ptrdiff_t i = 0; i < (ptrdiff_t)lhs.len; i++) {
413  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
414  return false;
415  }
416  }
417 
418  return true;
419 }

594 inline static void Slice99_swap(Slice99 self, ptrdiff_t lhs, ptrdiff_t rhs, void *restrict backup) {
595  SLICE99_ASSERT(backup);
596 
597  SLICE99_MEMCPY((backup), Slice99_get(self, lhs), (self).item_size);
598  SLICE99_MEMCPY(Slice99_get(self, lhs), Slice99_get(self, rhs), (self).item_size);
599  SLICE99_MEMCPY(Slice99_get(self, rhs), (backup), (self).item_size);
600 }

615 inline static void Slice99_swap_with_slice(Slice99 self, Slice99 other, void *restrict backup) {
616  SLICE99_ASSERT(self.len == other.len);
617  SLICE99_ASSERT(self.item_size == other.item_size);
618 
619  for (ptrdiff_t i = 0; i < (ptrdiff_t)self.len; i++) {
620  SLICE99_MEMCPY(backup, Slice99_get(self, i), self.item_size);
621  SLICE99_MEMCPY(Slice99_get(self, i), Slice99_get(other, i), self.item_size);
622  SLICE99_MEMCPY(Slice99_get(other, i), backup, self.item_size);
623  }
624 }
