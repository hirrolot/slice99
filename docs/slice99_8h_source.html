428 inline static SLICE99_WARN_UNUSED_RESULT Slice99
429 Slice99_from_ptrdiff(void *start, void *end, size_t item_size) {
430  SLICE99_ASSERT(start);
431  SLICE99_ASSERT(end);
432 
433  const ptrdiff_t diff = (char *)end - (char *)start;
434 
435  SLICE99_ASSERT(diff >= 0);
436  SLICE99_ASSERT((size_t)diff % item_size == 0);
437 
438  return Slice99_new(start, item_size, (size_t)(diff / (ptrdiff_t)item_size));
439 }

559  return Slice99_size(lhs) != Slice99_size(rhs)

578 inline static SLICE99_WARN_UNUSED_RESULT bool
579 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
580  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
581  SLICE99_ASSERT(comparator);
582 
583  if (lhs.len != rhs.len) {
584  return false;
585  }
586 
587  for (ptrdiff_t i = 0; i < (ptrdiff_t)lhs.len; i++) {
588  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
589  return false;
590  }
591  }
592 
593  return true;
594 }

715 inline static void Slice99_swap(Slice99 self, ptrdiff_t lhs, ptrdiff_t rhs, void *restrict backup) {
716  SLICE99_ASSERT(backup);
717 
718  SLICE99_MEMCPY((backup), Slice99_get(self, lhs), (self).item_size);
719  SLICE99_MEMCPY(Slice99_get(self, lhs), Slice99_get(self, rhs), (self).item_size);
720  SLICE99_MEMCPY(Slice99_get(self, rhs), (backup), (self).item_size);
721 }

736 inline static void Slice99_swap_with_slice(Slice99 self, Slice99 other, void *restrict backup) {
737  SLICE99_ASSERT(self.len == other.len);
738  SLICE99_ASSERT(self.item_size == other.item_size);
739 
740  for (ptrdiff_t i = 0; i < (ptrdiff_t)self.len; i++) {
741  SLICE99_MEMCPY(backup, Slice99_get(self, i), self.item_size);
742  SLICE99_MEMCPY(Slice99_get(self, i), Slice99_get(other, i), self.item_size);
743  SLICE99_MEMCPY(Slice99_get(other, i), backup, self.item_size);
744  }
745 }

756 inline static void Slice99_reverse(Slice99 self, void *restrict backup) {
757  SLICE99_ASSERT(backup);
758 
759  for (ptrdiff_t i = 0; i < (ptrdiff_t)self.len / 2; i++) {
760  Slice99_swap(self, i, (ptrdiff_t)self.len - i - 1, backup);

779  SLICE99_ASSERT(i <= self.len);
