406 inline static SLICE99_WARN_UNUSED_RESULT bool
407 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
408  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
409  SLICE99_ASSERT(comparator);
410 
411  if (lhs.len != rhs.len) {
412  return false;
413  }
414 
415  for (ptrdiff_t i = 0; i < (ptrdiff_t)lhs.len; i++) {
416  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
417  return false;
418  }
419  }
420 
421  return true;
422 }
