564 inline static SLICE99_WARN_UNUSED_RESULT bool
565 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
566  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
567  SLICE99_ASSERT(comparator);
568 
569  if (lhs.len != rhs.len) {
570  return false;
571  }
572 
573  for (ptrdiff_t i = 0; i < (ptrdiff_t)lhs.len; i++) {
574  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
575  return false;
576  }
577  }
578 
579  return true;
580 }

723  SLICE99_ASSERT(self.len == other.len);
