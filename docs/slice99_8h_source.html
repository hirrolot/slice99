575 inline static SLICE99_WARN_UNUSED_RESULT bool
576 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
577  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
578  SLICE99_ASSERT(comparator);
579 
580  if (lhs.len != rhs.len) {
581  return false;
582  }
583 
584  for (ptrdiff_t i = 0; i < (ptrdiff_t)lhs.len; i++) {
585  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
586  return false;
587  }
588  }
589 
590  return true;
591 }

734  SLICE99_ASSERT(self.len == other.len);
