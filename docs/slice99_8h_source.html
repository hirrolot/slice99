306 inline static bool
307 Slice99_eq(Slice99 lhs, Slice99 rhs, int (*comparator)(const void *, const void *)) {
308  SLICE99_ASSERT(lhs.item_size == rhs.item_size);
309  SLICE99_ASSERT(comparator);
310 
311  if (lhs.len != rhs.len) {
312  return false;
313  }
314 
315  for (size_t i = 0; i < lhs.len; i++) {
316  if (comparator(Slice99_get(lhs, i), Slice99_get(rhs, i)) != 0) {
317  return false;
318  }
319  }
320 
321  return true;
322 }
